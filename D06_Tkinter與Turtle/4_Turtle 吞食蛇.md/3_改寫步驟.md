2. GameScreen：封裝了 `turtle.Screen()` 物件的類，用於建立和控制一個 `遊戲畫面` 。

    ```python
    class GameScreen:
        # 初始化 turtle.Screen() 物件
        def __init__(self):
            self.screen = turtle.Screen()
            self.screen.title("吞食蛇遊戲")
            self.screen.bgcolor("black")
            self.screen.setup(width=600, height=600)
            # 關閉自動刷新，用於控制遊戲的繪畫效率
            self.screen.tracer(0)
        
        # 更新，用於手動刷新遊戲畫面
        def update(self):
            self.screen.update()
        
        # 監聽鍵盤事件
        def listen(self):
            self.screen.listen()
        
        # 設定當特定鍵被按下時的回調函數
        def onkeypress(self, func, key):
            self.screen.onkeypress(func, key)
        
        # 啟動 turtle 圖形窗口的主循環
        def mainloop(self):
            self.screen.mainloop()
    ```

3. Snake：封裝了 `turtle.Turtle()` 物件的類，用於創建和控制一個 `吞食蛇` 的物件。

    ```python
    class Snake:
        # 初始化 turtle.Turtle() 物件
        def __init__(self):
            # 創建了一個 turtle.Turtle() 物件作為蛇的頭部
            self.head = turtle.Turtle()
            # 初始速度為 0
            self.head.speed(0)
            # 形狀
            self.head.shape("square")
            # 顏色
            self.head.color("white")
            # 抬起畫筆，避免畫出軌跡
            self.head.penup()
            # 初始位置，畫面正中央
            self.head.goto(0, 0)
            # 初始方向
            # 在 Snake 類中，direction 屬性用於追蹤蛇頭的當前移動方向。
            # 這個屬性的值之中，"up"、"down"、"left"、"right" 用來控制蛇的移動。
            # 將 direction 設置為 "stop" 表示停止移動。
            # 當 move 方法被調用時會檢查 self.head.direction 的值作為移動方向的依據。
            self.head.direction = "stop"
            # 初始化 Snake 類的一個屬性，這個列表用來儲存和追蹤蛇的各部位
            # 初始僅有吞食蛇的頭部
            self.segments = [self.head]

        # 控制蛇的移動
        def move(self):
            # 遍歷蛇的各個部分，除了頭部，因為頭部的位置是新指派的，無需在這個遍歷中被設置。
            # 遍歷中會使每個部分移動到前一個身體部分的位置，從而實現移動效果。
            for index in range(len(self.segments) - 1, 0, -1):
                x = self.segments[index - 1].xcor()
                y = self.segments[index - 1].ycor()
                self.segments[index].goto(x, y)
            # 以下透過 if-elif 指定方向來移動蛇頭，每次移動固定的距離20像素
            if self.head.direction == "up":
                self.head.sety(self.head.ycor() + 20)
            elif self.head.direction == "down":
                self.head.sety(self.head.ycor() - 20)
            elif self.head.direction == "left":
                self.head.setx(self.head.xcor() - 20)
            elif self.head.direction == "right":
                self.head.setx(self.head.xcor() + 20)

        # 以下四個方法會檢查是否為反方向移動，藉此避免碰撞。
        def go_up(self):
            if self.head.direction != "down":
                self.head.direction = "up"

        def go_down(self):
            if self.head.direction != "up":
                self.head.direction = "down"

        def go_left(self):
            if self.head.direction != "right":
                self.head.direction = "left"

        def go_right(self):
            if self.head.direction != "left":
                self.head.direction = "right"
        # 增加蛇的長度，
        def extend(self):
            # 創建一個新的 turtle.Turtle() 物件作為新的「蛇身段」
            new_segment = turtle.Turtle()
            # 速度
            new_segment.speed(0)
            # 形狀
            new_segment.shape("square")
            # 顏色
            new_segment.color("grey")
            # 提起畫筆
            new_segment.penup()
            # 將新的「身段」加入蛇體
            self.segments.append(new_segment)
    ```

4. Food：食物，封裝了 `turtle.Turtle()` 物件的類。

    ```python
    class Food:
        def __init__(self):
            self.food = turtle.Turtle()
            self.food.speed(0)
            self.food.shape("circle")
            self.food.color("red")
            self.food.penup()
            # 初始位置
            self.food.goto(0, 100)
        
        # 以隨機產生的座標刷新食物的位置
        def refresh(self):
            # 透過 random.randint 函數建立隨機數
            x = random.randint(-290, 290)
            y = random.randint(-290, 290)
            # 類的屬性 food 會移動到新的座標
            self.food.goto(x, y)
    ```

5. Scoreboard：計分板。

    ```python
    class Scoreboard:
        def __init__(self):
            self.score = 0
            self.display = turtle.Turtle()
            self.display.speed(0)
            self.display.shape("square")
            self.display.color("white")
            self.display.penup()
            # 隱藏 turtle 的箭頭游標
            # hideturtle 方法將 turtle 物件設置為不可見狀態
            # 作為計分板時，只會看到文本，而不會看到那個小箭頭。
            self.display.hideturtle()
            # 設置計分板的位置
            self.display.goto(0, 260)
            # turtle 中用以顯示文本的函數
            self.display.write(
                # 文本
                "Score: 0", 
                # 置中
                align="center", 
                font=("Courier", 24, "normal")
            )

        # 顯示當前分數
        def update_score(self):
            # 清除顯示
            self.display.clear()
            # 寫入新的文本
            self.display.write(
                "Score: {}".format(self.score), 
                align="center", 
                font=("Courier", 24, "normal")
            )

        # 重置分數
        def reset(self):
            # 重置為 0
            self.score = 0
            # 調用自訂函數來更新分數
            self.update_score()
    ```

6. Game：遊戲物件，整合吞食蛇遊戲的各個組件並控制遊戲流程的類。

    ```python
    class Game:
        def __init__(self):
            # 使用 GameScreen 類創建遊戲畫面
            self.screen = GameScreen()
            # 使用 Snake 類創建蛇的實體
            self.snake = Snake()
            # 創建食物的實體
            self.food = Food()
            # 創建計分板實體
            self.scoreboard = Scoreboard()
            # 預設暫停狀態，特別說明，False 代表的是暫停
            # 確實理解設定的內容即可，不用太在意文字語意
            self.is_paused = False
            # 添加 delay 属性，可以理解為遊戲畫面刷新或更新速度
            # 延遲越短速度越快，這函數需實測調整到最佳體感
            self.delay = 0.1
            # 添加一個遊戲的控制的函數並且在初始化時調用
            self.setup_controls()
        
        # 設置遊戲的鍵盤控制
        def setup_controls(self):
            # 建立監聽，使 turtle 的畫面開始響應鍵盤事件
            self.screen.listen()
            # 使用 onkeypress 方法將特定按鍵綁定到特定的函數
            # 兩個參數分別是要綁定的函數與映射的按鍵
            self.screen.onkeypress(self.snake.go_up, "w")
            self.screen.onkeypress(self.snake.go_down, "s")
            self.screen.onkeypress(self.snake.go_left, "a")
            self.screen.onkeypress(self.snake.go_right, "d")
            self.screen.onkeypress(self.toggle_pause, "p")

        # 切換暫停的狀態
        def toggle_pause(self):
            # 觸發時會反轉當前狀態
            self.is_paused = not self.is_paused

        # 這是用來執行主循環的函數
        def run(self):
            # 主程序是一個無限循環
            while True:
                # 每次都會調用 update 函數更新畫面
                self.screen.update()
                # 更新後檢查吞食蛇是否處於暫停移動狀態
                if self.is_paused:
                    # 假如暫停移動就跳出本次迴圈
                    continue
                # 否則就調用 move 函數開始移動
                self.snake.move()
                # 移動後更新畫面
                self.screen.update()  
                # 先移動再檢查吞食蛇與食物的相對位置，假如小於代表發生碰撞
                if self.snake.head.distance(self.food.food) < 20:
                    # 食物調用自訂的刷新函數更換隨機位置
                    self.food.refresh()
                    # 吃到食物，吞食蛇調用自訂的加長蛇身段的函數 extend
                    self.snake.extend()
                    # 吃到食物，分數 + 10
                    self.scoreboard.score += 10
                    # 計分板更新分數
                    # 可以考慮將 score += 10 寫成參數傳給 update_score 函數
                    # 不然一件事分作兩步驟會增添程序出錯機率
                    self.scoreboard.update_score()
                
                # 以絕對值判斷蛇頭是否超過邊界
                if abs(self.snake.head.xcor()) > 290 or abs(self.snake.head.ycor()) > 290:
                    # 超過時代表撞牆了，這裡並不設置穿越
                    # 停頓一秒
                    time.sleep(1)
                    # 蛇頭回到原點
                    self.snake.head.goto(0, 0)
                    # 重置方向為停止
                    self.snake.head.direction = "stop"
                    # 逐一取出蛇身段
                    for segment in self.snake.segments[1:]:
                        # 將身段快速移出畫面
                        segment.goto(1000, 1000)
                    # 將蛇頭放入身段的列表中
                    self.snake.segments = [self.snake.head]
                    # 計分板重置
                    self.scoreboard.reset()

                # 檢測蛇頭是否自撞身段
                # 逐一取出蛇頭以外的身段
                for segment in self.snake.segments[1:]:
                    # 判斷是否自撞
                    if segment.distance(self.snake.head) < 20:
                        # 假如自撞了，停頓一秒鐘
                        time.sleep(1)
                        # 將蛇頭移動到原點
                        self.snake.head.goto(0, 0)
                        # 初始狀態為停止
                        self.snake.head.direction = "stop"
                        # 遍歷身段並快速移出畫面
                        for segment in self.snake.segments[1:]:
                            segment.goto(1000, 1000)
                        # 重置吞食蛇僅剩頭部
                        self.snake.segments = [self.snake.head]
                        # 計分板重置
                        self.scoreboard.reset()
                        # 特別說明，這部分明顯與上個程式碼區塊重複了，應再度封裝
                
                # 依照 Game 類所設置的延遲時間運行
                # 同理，寫到這裡也會覺得這個延遲應該寫入 Game 的初始化參數，而不是作為類的內置常數
                time.sleep(self.delay)
    ```

        game = Game()

## 補充說明

_若想嘗試重構重複的程式碼區塊_

1. 封裝撞牆或自撞之後的重置片段。

    ```python
    class Game:
        # ... 其他方法代碼 ...
        # 在 Game 類中新建一個方法 reset_game，並將重複的重置功能加入
        def reset_game(self):
            time.sleep(1)
            self.snake.head.goto(0, 0)
            self.snake.head.direction = "stop"
            for segment in self.snake.segments[1:]:
                segment.goto(1000, 1000)
            self.snake.segments = [self.snake.head]
            self.scoreboard.reset()
    ```
    
    _調用_
    
    ```python

        # 然後在 run 函數中的撞牆與自撞區塊各自調用 reset_game 方法
        def run(self):
            while True:
                # 其他邏輯代碼 ...

                # 撞牆
                if abs(self.snake.head.xcor()) > 290 or abs(self.snake.head.ycor()) > 290:
                    # 重置
                    self.reset_game()

                # 自撞
                for segment in self.snake.segments[1:]:
                    if segment.distance(self.snake.head) < 20:
                        # 重置
                        self.reset_game()

                # 其他邏輯代碼 ...
    ```
