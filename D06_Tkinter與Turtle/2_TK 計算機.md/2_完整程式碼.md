# 簡易計算機程式碼


_標題改為 `簡易` 計算機_

<br>

```python
from tkinter import *
# 初始化Tkinter視窗
tk_calc = Tk()
# 設定背景顏色和邊界寬度
tk_calc.configure(bg="#293C4A", bd=10)  
# 設定標題
tk_calc.title("簡易計算機")  


'''全局變數'''
calc_operator = ""  

# 兩種按鍵顏色模試
# 樣式一
button_params_main = {
    'bd': 5, 
    'fg': '#BBB', 
    'bg': '#3C3636', 
    'font': ('sans-serif', 20, 'bold')
}
# 樣式二
'''
button_params_main = {
    'bd': 5, 
    'fg': '#000', 
    'bg': '#BBB', 
    'font': ('sans-serif', 20, 'bold')
}
'''
# AC 按鍵使用不同顏色
button_params_AC = {
    'bd': 5, 
    'fg': '#000', 
    'bg': '#db701f', 
    'font': ('sans-serif', 20, 'bold')
}

'''實作函數'''
# 按鈕點擊事件：將字符加入到結果顯示屏幕中
def button_click(char):
    global calc_operator
    calc_operator += str(char)
    text_input.set(calc_operator)

# 計算結果：等號
def button_equal():
    global calc_operator
    try:
        # 評估運算式並返回結果
        calc_operator = str(eval(calc_operator))  
    # 處理除以零的錯誤
    except ZeroDivisionError:  
        calc_operator = "錯誤：除以零"
    # 處理其他所有錯誤
    except Exception:  
        calc_operator = "錯誤：無效輸入"
    text_input.set(calc_operator)

# 清除顯示屏所有內容
def button_clear_all():
    # 除了清除屏幕顯示內容，更要記得清除變數
    global calc_operator
    calc_operator = ""
    # 清除屏幕
    text_input.set("")

'''控件變數（widget variable)'''
# 如 StringVar ...
# 創建一個 StringVar 類型的字串變數來儲存結果運算框的文本
text_input = StringVar() 

'''建立控件物件'''
# 創建數字按鈕
for i in range(9):
    # 使用迴圈建立九個數字鍵
    Button(
        tk_calc, 
        button_params_main, text=str(i + 1),
        command=lambda num=i + 1: button_click(str(num))
    ).grid(
        row=(i // 3) + 1, 
        column=i % 3, 
        sticky="nsew"
    )
# 設定輸入框
Entry(
    tk_calc, 
    font=('sans-serif', 20, 'bold'), 
    textvariable=text_input,
    bd=5, 
    insertwidth=5, 
    bg='#BBB', 
    justify='right'
).grid(
    row=0, 
    column=0,
    columnspan=5,
    padx=10, 
    pady=15
)
# 創建按鈕0
Button(
    tk_calc, 
    button_params_main, 
    text='0', 
    command=lambda: button_click('0')
).grid(
    row=4, 
    column=1, 
    sticky="nsew"
)

# 加減乘除的按鈕物件：+-*/ 
# +
Button(
    tk_calc, 
    button_params_main, 
    text='+', 
    command=lambda: button_click('+')
).grid(
    row=1, 
    column=3, 
    sticky="nsew"
)
# -
Button(
    tk_calc, 
    button_params_main, 
    text='-', 
    command=lambda: button_click('-')
).grid(
    row=2,
    column=3, 
    sticky="nsew"
)
# *
Button(
    tk_calc, 
    button_params_main, 
    text='*', 
    command=lambda: button_click('*')
).grid(
        row=3, 
        column=3, 
        sticky="nsew"
    )
# /
Button(
    tk_calc, 
    button_params_main, 
    text='/', 
    command=lambda: button_click('/')
).grid(
    row=4, 
    column=3, 
    sticky="nsew"
)
# 創建小數點按鈕
Button(
    tk_calc, 
    button_params_main, 
    text='.', 
    command=lambda: button_click('.')
).grid(
    row=4, 
    column=0, 
    sticky="nsew"
)
# 等號按鈕
Button(
    tk_calc, 
    button_params_main, 
    text='=', 
    command=button_equal
).grid(
    row=4, 
    column=2, 
    sticky="nsew"
)
# AC
Button(
    tk_calc, 
    button_params_AC,
    text='AC', 
    command=button_clear_all
).grid(
    row=1, 
    column=4, 
    rowspan=4,
    sticky="nsew"
)

# tk 進入 loop 運作
tk_calc.mainloop()
```
