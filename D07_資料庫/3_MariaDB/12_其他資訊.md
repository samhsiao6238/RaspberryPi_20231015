## 範例

1. 導入庫。

  ```python
  import pandas as pd
  import requests
  import mariadb
  import sys
  ```

<br>

2. 從網路取得所需的數據資料。

  ```python
  # 設定目標日期
  _TARGET_DATA = '20230928'

  # 抓取股市資訊
  url = f'https://www.twse.com.tw/rwd/zh/afterTrading/MI_INDEX?date={_TARGET_DATA}&type=ALL&response=csv'
  res = requests.get(url)
  data = res.text
  ```

<br>

3. 資料整理與數據清洗。

  ```python
  # 切割與整理資料
  s_data = data.split('\n')
  # 用來存放
  output = []
  for d in s_data:
      _d = d.split('","')
      if len(_d) == 16 and not _d[0].startswith('='):
          # 移除全部的 \r ''
          output.append([ele.replace('",\r','').replace('"','') for ele in _d])

  # 轉檔案：轉成 DataFrame
  df = pd.DataFrame(output[1:], columns=output[0])

  # 將 DataFrame 中的數字型欄位轉換為數字格式
  numeric_columns = [
    '成交股數', 
    '成交筆數', 
    '成交金額', 
    '開盤價', 
    '最高價', 
    '最低價', 
    '收盤價', 
    '漲跌價差', 
    '最後揭示買價', 
    '最後揭示買量', 
    '最後揭示賣價', 
    '最後揭示賣量', 
    '本益比'
  ]

  # 遍歷 numeric_columns
  for col in numeric_columns:
      # 使用 replace 將字典映射的 key 換成 value
      df[col] = df[col].replace(
        {',': '', '--': None}, 
        # 將字典中的 key 視為正則表達式
        regex=True
      )
      # 假如符合這個條件，就將值轉換為數值型態
      if col in ['成交股數', '成交筆數', '成交金額', '開盤價', '最高價', '最低價', '收盤價', '漲跌價差', '最後揭示買價', '最後揭示賣價', '本益比']:
          # 使用 pandas 的函數 to_numeric，將一列數據轉換為數值型態
          # 設置為 coerce 表示如果在轉換過程中無法解析為數值，就將值設置為 NaN
          df[col] = pd.to_numeric(df[col], errors='coerce')
      else:
          # 暫且保留原有內容與格式
          pass

  # 確保所有 NaN 值被轉換為 None
  for col in df.columns:
      '''
      下面這段表達式中的 lamba 等價於：
      if pd.isna(x) == True:
        x = None
      '''
      df[col] = df[col].map(lambda x: None if pd.isna(x) else x)
  ```

<br>

4. 連線資料庫。

  ```python
  # 連接到 MariaDB
  try:
      conn = mariadb.connect(
          user="sam6238",
          password="sam112233",
          host="localhost",
          port=3306,
          database="mydatabase"
      )
  except mariadb.Error as e:
      print(f"連接 MariaDB 發生錯誤：{e}")
      sys.exit(1)

  # 創建 cursor，提供後續操作資料庫所用
  cur = conn.cursor()
  ```

<br>

5. 建立資料表。

  ```python
  # 創建表格
  try:
      # 根据提供的字段创建表格，字段类型可能需要根据实际情况调整
      cur.execute("""
          CREATE TABLE IF NOT EXISTS stock_TW (
              date DATE,
              證券代號 VARCHAR(10),
              證券名稱 VARCHAR(50),
              成交股數 BIGINT,
              成交筆數 INT,
              成交金額 BIGINT,
              開盤價 FLOAT,
              最高價 FLOAT,
              最低價 FLOAT,
              收盤價 FLOAT,
              漲跌 VARCHAR(5),
              漲跌價差 FLOAT,
              最後揭示買價 FLOAT,
              最後揭示買量 INT,
              最後揭示賣價 FLOAT,
              最後揭示賣量 INT,
              本益比 FLOAT
          )
      """)
  except mariadb.Error as e:
      print(f"Error creating table: {e}")
  ```

<br>

6. 寫入資料庫。

  ```python
  # 定義插入數據的 SQL 語句
  insert_statement = """
  INSERT INTO stock_TW (
      date, 證券代號, 證券名稱, 成交股數, 成交筆數, 成交金額, 開盤價, 最高價, 最低價, 收盤價, 漲跌, 漲跌價差, 最後揭示買價, 最後揭示買量, 最後揭示賣價, 最後揭示賣量, 本益比
  ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  """

  # 將 df 寫入到資料庫
  for index, row in df.iterrows():
      try:
          cur.execute(insert_statement, (
              _TARGET_DATA,
              index,
              row['證券名稱'],
              row['成交股數'],
              row['成交筆數'],
              row['成交金額'],
              row['開盤價'],
              row['最高價'],
              row['最低價'],
              row['收盤價'],
              row['漲跌(+/-)'],
              row['漲跌價差'],
              row['最後揭示買價'],
              row['最後揭示買量'],
              row['最後揭示賣價'],
              row['最後揭示賣量'],
              row['本益比']
          ))
      except mariadb.Error as e:
          print(f"{index}, row['證券名稱'], 錯誤: {e}")

  # 提交事务
  conn.commit()
  ```

<br>

7. 讀取資料庫。

  ```python
  # 讀取最新 20 筆資料
  try:
      cur.execute("SELECT * FROM stock_TW ORDER BY date DESC LIMIT 20")
      # 查看一下
      for row in cur:
        print(row)
  except mariadb.Error as e:
      print(f"錯誤: {e}")
  ```

<br>

8. 檢查資料。

  ```python
  # 運行查詢以檢查含有 NULL 值的行
  try:
      cur.execute("SELECT * FROM stock_TW WHERE date IS NULL OR 證券代號 IS NULL OR 證券名稱 IS NULL OR 成交股數 IS NULL OR 成交筆數 IS NULL OR 成交金額 IS NULL OR 開盤價 IS NULL OR 最高價 IS NULL OR 最低價 IS NULL OR 收盤價 IS NULL OR 漲跌 IS NULL OR 漲跌價差 IS NULL OR 最後揭示買價 IS NULL OR 最後揭示買量 IS NULL OR 最後揭示賣價 IS NULL OR 最後揭示賣量 IS NULL OR 本益比 IS NULL")
      rows_with_null = cur.fetchall()
      for row in rows_with_null:
          print(row)
  except mariadb.Error as e:
      print(f"錯誤: {e}")
  ```

<br>

9. 關閉連線。

  ```python
  # 關閉連接
  cur.close()
  conn.close()
  ```
