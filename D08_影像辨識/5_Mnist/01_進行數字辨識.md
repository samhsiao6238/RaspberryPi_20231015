## 範例

1. 建立虛擬環境。

    ```bash
    python -m venv envMnist
    ```

<br>

2. 安裝庫。

    ```bash
    pip install tensorflow opencv-python opencv_contrib_python
    ```

<br>

3. 導入必要庫。

    ```python
    import cv2
    import numpy as np
    # 從 TensorFlow Keras 導入模型加載功能
    from tensorflow.keras.models import load_model
    # 導入 sys 函式庫，用於系統相關操作
    import sys
    ```

<br>


4. 載入模型與攝像頭。

    ```python
    # 載入模型
    try:
        model = load_model("mnist_model.h5")
        print("模型載入成功。")
    except OSError as e:
        print("模型載入失敗：", e)
        # 退出程序
        sys.exit(1)

    # 啟用攝影鏡頭
    cap = cv2.VideoCapture(0)
    print("啟動鏡頭 ...")

    # 如果攝像頭讀取失敗
    if not cap.isOpened():
        print("無法開啟攝像頭")
        # 退出程序
        sys.exit(1)
    ```

<br>


5. 主程序：主程序將是在一個 `while` 迴圈中運行直到退出，所以先寫上 `while` 再來逐步填入。

    ```python
    while True:
        pass
    ```

<br>


6. 擷取影像。

    ```python
    '''擷取影像：每一次回圈'''
    # 從攝影鏡頭讀取影像
    success, img = cap.read()
    # 如果讀取失敗
    if not success:
        print("無法取得影像")
        # 中斷迴圈
        break

    # 若要加快運行速度，可縮小影像尺寸
    img = cv2.resize(img, (450, 300))

    # 擷取區域的座標與大小，可自行調整座標與大小
    x, y, w, h = 300, 150, 100, 100

    # 擷取辨識的區域
    img_num = img[y: y + h, x: x + w]
    img_num = cv2.cvtColor(img_num, cv2.COLOR_BGR2GRAY)
    _, img_num = cv2.threshold(
        img_num, 127, 255, cv2.THRESH_BINARY_INV
    )
    # 調整影像大小以符合模型輸入
    img_num = cv2.resize(img_num, (28, 28))
    # 將影像數據類型轉換為 float32，並進行歸一化
    img_num = img_num.astype("float32") / 255
    # 擴展影像維度以符合模型輸入要求
    img_num = np.expand_dims(img_num, axis=0)
    img_num = np.expand_dims(img_num, axis=-1)
    ```

<br>


7. 進行辨識。

    ```python
    # 使用模型進行預測
    predictions = model.predict(img_num)
    # 獲得預測結果中概率最高的類別
    num = np.argmax(predictions, axis=1)[0]
    # 將準確率轉換為百分比
    confidence = np.max(predictions) * 100
    ```

<br>


8. 文字輸出。

    ```python
    # 文字內容
    text = f"Number: {num}, Confidence: {confidence:.2f}%"
    # 文字位置，在畫面左上角顯示
    org = (10, 30)
    # 字體
    fontFace = cv2.FONT_HERSHEY_SIMPLEX
    # 字體大小
    fontScale = 0.7
    # 字體顏色：白色文字
    color = (255, 255, 255)
    # 字體粗細
    thickness = 2
    # 線條類型
    lineType = cv2.LINE_AA
    # 在影像上顯示文字
    cv2.putText(
        img, text, org, fontFace, fontScale, color, thickness, lineType
    )
    ```

<br>


9. 辨識區域設置。

    ```python
    # 繪製辨識的區域，提示使用者將辨識物件置於框線中
    cv2.rectangle(img, (x, y), (x + w, y + h), (0, 0, 255), 3)
    ```

<br>


10. 影像輸出。

    ```python
    # 顯示處理後的影像並加上標題
    cv2.imshow("Minist ", img)
    ```

<br>


11. 退出機制。

    ```python
    # 按下 q 鍵停止
    if cv2.waitKey(1) == ord("q"):
        break
    ```

<br>



12. 釋放資源：結束 `while` 回圈之後代表程序結束。

    ```python
    cap.release()
    cv2.destroyAllWindows()
    ```

## 完整範例

1. 程式碼。

    ```python
    '''載入必要庫'''
    import cv2
    import numpy as np
    # 從 TensorFlow Keras 導入模型加載功能
    from tensorflow.keras.models import load_model
    # 導入 sys 函式庫，用於系統相關操作
    import sys

    '''載入模型與攝像頭'''
    # 載入模型
    try:
        model = load_model("mnist_model.h5")
        print("模型載入成功。")
    except OSError as e:
        print("模型載入失敗：", e)
        # 退出程序
        sys.exit(1)

    # 啟用攝影鏡頭
    cap = cv2.VideoCapture(0)
    print("啟動鏡頭 ...")

    # 如果攝像頭讀取失敗
    if not cap.isOpened():
        print("無法開啟攝像頭")
        # 退出程序
        sys.exit(1)

    '''主程序'''
    while True:
        '''擷取影像：每一次回圈'''
        # 從攝影鏡頭讀取影像
        success, img = cap.read()
        # 如果讀取失敗
        if not success:
            print("無法取得影像")
            # 中斷迴圈
            break

        # 若要加快運行速度，可縮小影像尺寸
        img = cv2.resize(img, (450, 300))

        # 擷取區域的座標與大小，可自行調整座標與大小
        x, y, w, h = 300, 150, 100, 100

        # 擷取辨識的區域
        img_num = img[y: y + h, x: x + w]
        img_num = cv2.cvtColor(img_num, cv2.COLOR_BGR2GRAY)
        _, img_num = cv2.threshold(
            img_num, 127, 255, cv2.THRESH_BINARY_INV
        )
        # 調整影像大小以符合模型輸入
        img_num = cv2.resize(img_num, (28, 28))
        # 將影像數據類型轉換為 float32，並進行歸一化
        img_num = img_num.astype("float32") / 255
        # 擴展影像維度以符合模型輸入要求
        img_num = np.expand_dims(img_num, axis=0)
        img_num = np.expand_dims(img_num, axis=-1)

        '''
        進行辨識
        '''
        # 使用模型進行預測
        predictions = model.predict(img_num)
        # 獲得預測結果中概率最高的類別
        num = np.argmax(predictions, axis=1)[0]
        # 將準確率轉換為百分比
        confidence = np.max(predictions) * 100

        '''
        文字輸出設置
        '''
        # 文字內容
        text = f"Number: {num}, Confidence: {confidence:.2f}%"
        # 文字位置，在畫面左上角顯示
        org = (10, 30)
        # 字體
        fontFace = cv2.FONT_HERSHEY_SIMPLEX
        # 字體大小
        fontScale = 0.7
        # 字體顏色：白色文字
        color = (255, 255, 255)
        # 字體粗細
        thickness = 2
        # 線條類型
        lineType = cv2.LINE_AA
        # 在影像上顯示文字
        cv2.putText(
            img, text, org, fontFace, fontScale, color, thickness, lineType
        )

        '''
        畫面輸出
        '''
        # 繪製辨識的區域，提示使用者將辨識物件置於框線中
        cv2.rectangle(img, (x, y), (x + w, y + h), (0, 0, 255), 3)
        # 顯示處理後的影像並加上標題
        cv2.imshow("Minist ", img)

        '''
        退出機制
        '''
        # 按下 q 鍵停止
        if cv2.waitKey(1) == ord("q"):
            break

    '''
    釋放資源
    '''
    # 釋放資源並關閉視窗
    cap.release()
    cv2.destroyAllWindows()
    ```
