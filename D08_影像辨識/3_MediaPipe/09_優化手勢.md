## 說明

1. 畫面左右翻轉，呈現鏡像顯示。

    ```python

    ```

2. 封裝繪製手勢辨識結果的函數。
    
    ```python
    # 在指定位置繪製手勢辨識結果
    def draw_text(
        image, text, position, font, font_scale,
        font_color, font_thickness, line_type

        cv2.putText(
            image, text, position, font, font_scale,
            font_color, font_thickness, line_type
        )
    ```

<br>

3. 添加判斷是左手還是右手比出手勢：依據手掌根部在畫面的相對位置來判斷。

    ```python
    # 決定顯示位置，根據需要調整寬度減去的值
    # 如果手掌根部的 x 座標在畫面左邊則為左手
    if finger_points[0][0] < w / 2:
        # 左手，顯示在左側
        position = (30, 200 + hand_index * 50)
    # 否則是右手，顯示在右側
    else:    
        position = (w - 600, 200 + hand_index * 50)  
    ```

<br>

4. 添加特定手勢可以執行命令，這裡是以兩隻手比出 `Rock` 手勢的時候，結束程序。

    ```python
    # 在 while 回圈一開始的時候地方
    # 添加一個變數追蹤「ROCK」手勢次數，每次回圈都要歸零
    with mp_hands.Hands(...) as hands:
        # ... 其他代碼保持不變 ...
        #
        rock_hand_count = 0

        # 在 finger_points 區塊得到 text 的之後，檢查並且計算次數
        if finger_points:
            # 其他不變 ...
            # 取得 text
            text = hand_pos(finger_angle)
            # 檢查是否是「ROCK」手勢
            if text == "ROCK!":
                rock_hand_count += 1
        # 其他不變 ...
        
        # 接著在迴圈的最後
        # 檢查是否有兩隻手都是「ROCK」手勢
        if rock_hand_count >= 2:
            # 可以添加其他的通知
            break
    ```

## 範例

1. 完整程式碼。

    ```python
    """
    可偵測1-9、OK、Good以及中指手勢
    """
    import cv2
    import mediapipe as mp
    import math
    import sys

    mp_drawing = mp.solutions.drawing_utils
    mp_drawing_styles = mp.solutions.drawing_styles
    mp_hands = mp.solutions.hands


    # 在指定位置繪製手勢辨識結果
    def draw_text(
        image, text, position, font, font_scale,
        font_color, font_thickness, line_type
    ):
        cv2.putText(
            image, text, position, font, font_scale,
            font_color, font_thickness, line_type
        )


    # 根據兩點的座標，計算角度
    def vector_2d_angle(v1, v2):
        v1_x = v1[0]
        v1_y = v1[1]
        v2_x = v2[0]
        v2_y = v2[1]
        try:
            # 透過計算兩個向量的點積和各自長度來求出它們之間的夾角
            # 這樣單行表達易讀性太低，最好優化分拆
            angle = math.degrees(
                math.acos(
                    (v1_x * v2_x + v1_y * v2_y)
                    / (((v1_x**2 + v1_y**2) ** 0.5)
                        * ((v2_x**2 + v2_y**2) ** 0.5))
                ))
        except Exception as e:
            # 這個例外的發生是兩向量完全相反的情況下
            # 所以它們之間的角度是180度
            angle = 180
            print(f"捕捉例外，推判並設定夾角為 180：{e}")
        return angle


    # 根據傳入的 21 個節點座標，得到該手指的角度
    def hand_angle(hand_):
        angle_list = []
        # thumb 大拇指角度
        angle = vector_2d_angle(
            (
                (int(hand_[0][0]) - int(hand_[2][0])),
                (int(hand_[0][1]) - int(hand_[2][1]))
            ),
            (
                (int(hand_[3][0]) - int(hand_[4][0])),
                (int(hand_[3][1]) - int(hand_[4][1]))
            ),
        )
        angle_list.append(angle)
        # index 食指角度
        angle = vector_2d_angle(
            (
                (int(hand_[0][0]) - int(hand_[6][0])),
                (int(hand_[0][1]) - int(hand_[6][1]))
            ),
            (
                (int(hand_[7][0]) - int(hand_[8][0])),
                (int(hand_[7][1]) - int(hand_[8][1]))
            ),
        )
        angle_list.append(angle)
        # middle 中指角度
        angle = vector_2d_angle(
            (
                (int(hand_[0][0]) - int(hand_[10][0])),
                (int(hand_[0][1]) - int(hand_[10][1])),
            ),
            (
                (int(hand_[11][0]) - int(hand_[12][0])),
                (int(hand_[11][1]) - int(hand_[12][1])),
            ),
        )
        angle_list.append(angle)
        # ring 無名指角度
        angle = vector_2d_angle(
            (
                (int(hand_[0][0]) - int(hand_[14][0])),
                (int(hand_[0][1]) - int(hand_[14][1])),
            ),
            (
                (int(hand_[15][0]) - int(hand_[16][0])),
                (int(hand_[15][1]) - int(hand_[16][1])),
            ),
        )
        angle_list.append(angle)
        # pink 小拇指角度
        angle = vector_2d_angle(
            (
                (int(hand_[0][0]) - int(hand_[18][0])),
                (int(hand_[0][1]) - int(hand_[18][1])),
            ),
            (
                (int(hand_[19][0]) - int(hand_[20][0])),
                (int(hand_[19][1]) - int(hand_[20][1])),
            ),
        )
        angle_list.append(angle)
        return angle_list


    # 根據手指角度的串列內容，返回對應的手勢名稱
    def hand_pos(finger_angle):
        # 大拇指角度
        f1 = finger_angle[0]
        # 食指角度
        f2 = finger_angle[1]
        # 中指角度
        f3 = finger_angle[2]
        # 無名指角度
        f4 = finger_angle[3]
        # 小拇指角度
        f5 = finger_angle[4]

        # 小於 50 表示手指伸直，大於等於 50 表示手指捲縮
        if f1 < 50 and f2 >= 50 and f3 >= 50 and f4 >= 50 and f5 >= 50:
            return "Good"
        elif f1 >= 50 and f2 >= 50 and f3 < 50 and f4 >= 50 and f5 >= 50:
            return "Don't do that."
        elif f1 < 50 and f2 < 50 and f3 >= 50 and f4 >= 50 and f5 < 50:
            return "ROCK"
        elif f1 >= 50 and f2 >= 50 and f3 >= 50 and f4 >= 50 and f5 >= 50:
            return "0"
        elif f1 >= 50 and f2 >= 50 and f3 >= 50 and f4 >= 50 and f5 < 50:
            return "Pinky"
        elif f1 >= 50 and f2 < 50 and f3 >= 50 and f4 >= 50 and f5 >= 50:
            return "1"
        elif f1 >= 50 and f2 < 50 and f3 < 50 and f4 >= 50 and f5 >= 50:
            return "2"
        elif f1 >= 50 and f2 >= 50 and f3 < 50 and f4 < 50 and f5 < 50:
            return "ok"
        elif f1 < 50 and f2 >= 50 and f3 < 50 and f4 < 50 and f5 < 50:
            return "ok"
        elif f1 >= 50 and f2 < 50 and f3 < 50 and f4 < 50 and f5 > 50:
            return "3"
        elif f1 >= 50 and f2 < 50 and f3 < 50 and f4 < 50 and f5 < 50:
            return "4"
        elif f1 < 50 and f2 < 50 and f3 < 50 and f4 < 50 and f5 < 50:
            return "5"
        elif f1 < 50 and f2 >= 50 and f3 >= 50 and f4 >= 50 and f5 < 50:
            return "6"
        elif f1 < 50 and f2 < 50 and f3 >= 50 and f4 >= 50 and f5 >= 50:
            return "7"
        elif f1 < 50 and f2 < 50 and f3 < 50 and f4 >= 50 and f5 >= 50:
            return "8"
        elif f1 < 50 and f2 < 50 and f3 < 50 and f4 < 50 and f5 >= 50:
            return "9"
        else:
            return ""


    # 讀取攝影機
    cap = cv2.VideoCapture(0)
    # 印出文字的字型
    fontFace = cv2.FONT_HERSHEY_SIMPLEX
    # 印出文字的邊框
    lineType = cv2.LINE_AA

    # 初始化手掌偵測
    with mp_hands.Hands(
        model_complexity=0,
        min_detection_confidence=0.5,
        min_tracking_confidence=0.5
    ) as hands:
        if not cap.isOpened():
            print("無法找到攝像頭")
            sys.exit(1)

        # 獲取攝影機原始分辨率
        original_width = cap.get(cv2.CAP_PROP_FRAME_WIDTH)
        original_height = cap.get(cv2.CAP_PROP_FRAME_HEIGHT)
        # 計算比率
        aspect_ratio = original_height / original_width

        # 影像尺寸
        w = 1080
        h = int(w * aspect_ratio)

        # 用來記錄每隻手的手勢狀態
        hand_gestures = {}

        while True:
            # 讀取畫面
            success, img = cap.read()
            # 縮小尺寸，加快處理效率
            img = cv2.resize(img, (w, h))
            if not success:
                print("無法獲取畫面")
                break
            # 添加鏡像效果，水平翻轉影像
            img = cv2.flip(img, 1)

            # 轉換成 RGB 色彩
            img2 = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
            # 偵測手勢
            results = hands.process(img2)
            # 添加一個變數
            # hand_index = 0
            if results.multi_hand_landmarks:
                for hand_landmarks in results.multi_hand_landmarks:
                    # 記錄手指節點座標的串列
                    finger_points = []
                    for i in hand_landmarks.landmark:
                        # 將 21 個節點換算成座標，記錄到 finger_points
                        x = i.x * w
                        y = i.y * h
                        finger_points.append((x, y))
                    if finger_points:
                        # 計算手指角度，回傳長度為 5 的串列
                        finger_angle = hand_angle(finger_points)
                        # 印出角度 ( 有需要就開啟註解 )
                        # print(finger_angle)
                        # 取得手勢所回傳的內容
                        text = hand_pos(finger_angle)

                        # 根據手的序號決定顯示位置
                        # 決定顯示位置：如果手掌根部的 x 座標（finger_points[0][0]）較小，則為左手
                        if finger_points[0][0] < w / 2:
                            # 左手，顯示在左側
                            position = (30, 150)
                        else:
                            # 右手，顯示在右側
                            # 根據需要調整寬度減去的值
                            position = (int(w / 2), 150)

                        # 決定是左手還是右手
                        hand_side = "Left" if finger_points[0][0] \
                            < w / 2 else "Right"

                        # 更新手勢狀態
                        hand_gestures[hand_side] = text
                        # 顯示偵測結果在螢幕上
                        draw_text(
                            img, text, position, fontFace, 5,
                            (255, 255, 255), 10, lineType
                        )

            # 檢查是否兩隻手都是 'ROCK' 手勢
            if len(hand_gestures) == 2 and all(
                hand == "ROCK" for hand in hand_gestures.values()
            ):
                # 如果是，則退出程序
                break

            # 顯示影像並設置標題
            cv2.imshow("Optimize Gestures", img)
            # 檢查是否有按下'ESC'、'q'鍵或關閉視窗
            key = cv2.waitKey(1) & 0xFF
            if (
                key == 27
                or key == ord("q")
                or cv2.getWindowProperty(
                    "Optimize Gestures", cv2.WND_PROP_VISIBLE
                ) < 1
            ):
                break

    cap.release()
    cv2.destroyAllWindows()

    ```
