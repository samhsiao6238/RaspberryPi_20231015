# 物體偵測

<br>

## 說明

1. 可參考 [官網文件](https://github.com/google/mediapipe/blob/master/docs/solutions/objectron.md) 。

<br>

2. 這個範例包含兩個部分，首先是偵測預先提供的相片集，接著是偵測影像。


<br>

## 範例

1. 完整程式碼。

    ```python
    """
    偵測物體
    參考官網
    https://github.com/google/mediapipe/blob/master/docs/solutions/objectron.md
    """
    import cv2
    import mediapipe as mp
    import os

    # 初始化mediapipe繪圖工具和物體偵測工具
    mp_drawing = mp.solutions.drawing_utils
    mp_objectron = mp.solutions.objectron

    # 設定要處理的靜態圖像列表
    IMAGE_FILES = [
        "shoe_01.jpg",
        "shoe_02.jpg",
        "shoe_03.jpg",
        "shoe_04.jpg",
        "shoe_05.jpg",
    ]

    # 配置並初始化mediapipe物體偵測（Objectron）模型
    with mp_objectron.Objectron(
        static_image_mode=True,
        max_num_objects=5,
        min_detection_confidence=0.5,
        model_name="Shoe",
    ) as objectron:
        # 對每個圖像檔案進行處理
        for idx, file in enumerate(IMAGE_FILES):
            image = cv2.imread(file)
            # 將圖像從 BGR 轉換成 RGB 格式，並使用 Objectron 進行處理
            results = objectron.process(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))

            # 繪製檢測到的物體的邊界框
            if not results.detected_objects:
                print(f"No box landmarks detected on {file}")
                continue
            print(f"Box landmarks of {file}:")
            annotated_image = image.copy()
            #
            for detected_object in results.detected_objects:
                # 使用mediapipe繪圖工具繪製標記
                mp_drawing.draw_landmarks(
                    annotated_image,
                    detected_object.landmarks_2d,
                    mp_objectron.BOX_CONNECTIONS,
                )
                # 使用mediapipe繪圖工具繪製坐標軸
                mp_drawing.draw_axis(
                    annotated_image,
                    detected_object.rotation,
                    detected_object.translation
                )
                # 將處理後的圖像保存到指定路徑
                data_folder = "data"
                if not os.path.exists(data_folder):
                    os.makedirs(data_folder)

                try:
                    cv2.imwrite(
                    "data/annotated_image" + str(idx) + ".png", annotated_image
                    )
                except Exception as e:
                    print(f"發生錯誤：{e}")

    # 處理攝像頭輸入
    cap = cv2.VideoCapture(0)
    with mp_objectron.Objectron(
        static_image_mode=False,
        max_num_objects=5,
        min_detection_confidence=0.5,
        min_tracking_confidence=0.99,
        model_name="Shoe",
    ) as objectron:
        while cap.isOpened():
            success, image = cap.read()
            if not success:
                print("Ignoring empty camera frame.")
                # 如果加載視頻，請使用 break 而不是 continue
                continue

            # 將圖像設為不可寫，以提升性能
            image.flags.writeable = False
            image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
            # 使用Objectron處理圖像
            results = objectron.process(image)

            # 繪製檢測到的物體的邊界框
            image.flags.writeable = True
            image = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)
            if results.detected_objects:
                for detected_object in results.detected_objects:
                    mp_drawing.draw_landmarks(
                        image,
                        detected_object.landmarks_2d,
                        mp_objectron.BOX_CONNECTIONS
                    )
                    mp_drawing.draw_axis(
                        image,
                        detected_object.rotation,
                        detected_object.translation
                    )
            # 水平翻轉圖像以進行自拍視角展示
            cv2.imshow("MediaPipe Objectron", cv2.flip(image, 1))
            # 按下 Q 退出
            if cv2.waitKey(5) == ord('q'):
                break

    # 釋放
    cap.release()
    ```
