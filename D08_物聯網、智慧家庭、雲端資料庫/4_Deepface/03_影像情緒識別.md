# 即時影像情緒識別

<br>

## 範例

1. 完整程式碼。

    ```python
    """
    即時影像顯示情緒
    """
    import cv2
    from deepface import DeepFace
    import numpy as np
    from PIL import ImageFont, ImageDraw, Image

    # Mapping
    text_obj = {
        "angry": "生氣",
        "disgust": "噁心",
        "fear": "害怕",
        "happy": "開心",
        "sad": "難過",
        "surprise": "驚訝",
        "neutral": "正常",
    }


    # 定義加入文字函式
    def putText(x, y, text, size=50, color=(255, 255, 255)):
        global img
        # 字型
        fontpath = "NotoSansTC-VariableFont_wght.ttf"
        # 定義字型與文字大小
        font = ImageFont.truetype(fontpath, size)
        # 轉換成 PIL 影像物件
        imgPil = Image.fromarray(img)
        # 定義繪圖物件
        draw = ImageDraw.Draw(imgPil)
        # 加入文字
        draw.text((x, y), text_obj[text], fill=color, font=font)
        # 轉換成 np.array
        img = np.array(imgPil)


    cap = cv2.VideoCapture(0)

    if not cap.isOpened():
        print("無法開啟攝像頭")
        exit()
    while True:
        ret, frame = cap.read()
        if not ret:
            print("無法取得畫面")
            break
        img = cv2.resize(frame, (384, 240))
        try:
            analyze = DeepFace.analyze(img, actions=["emotion"])
            # 取得情緒文字
            emotion = analyze[0]["dominant_emotion"]
            # 放入文字
            putText(0, 40, emotion)
        except Exception as e:
            print(f'發生錯誤：{e}')
        cv2.imshow("Example_05", img)

        if cv2.waitKey(5) == ord("q"):
            break
    cap.release()
    cv2.destroyAllWindows()
    ```
